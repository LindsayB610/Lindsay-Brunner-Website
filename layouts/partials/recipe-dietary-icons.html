{{/* Renders minimal dietary icons (dairy-free, vegetarian, vegan, gluten-free) as links to filtered recipe list. */}} {{/* Expects: dict "page" . "base" "/recipes/" */}} {{- $page := .page -}} {{- $base := default "/recipes/" .base -}} {{- $dietary := $page.Params.dietary -}} {{- if $dietary -}} {{- range $dietary -}} {{- $slug := . | urlize -}} {{- $label := . -}} {{- $url := printf "%s?diet=%s" (strings.TrimSuffix "/" $base) $slug -}} {{- if eq $slug "dairy-free" -}} {{- else if eq $slug "vegetarian" -}} {{- else if eq $slug "vegan" -}} {{- else if eq $slug "gluten-free" -}} {{- end -}} {{- end -}} {{- end -}}
