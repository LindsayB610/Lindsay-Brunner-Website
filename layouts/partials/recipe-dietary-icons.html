{{/* Renders minimal dietary labels (DF, V, VG, GF) as links to filtered recipe list. */}} {{/* Expects: dict "page" . "base" "/recipes/" */}} {{- $page := .page -}} {{- $base := default "/recipes/" .base -}} {{- $dietary := $page.Params.dietary -}} {{- if $dietary -}} {{- range $dietary -}} {{- $slug := . | urlize -}} {{- $label := . -}} {{- $abbr := cond (eq $slug "dairy-free") "DF" (cond (eq $slug "vegetarian") "V" (cond (eq $slug "vegan") "VG" (cond (eq $slug "gluten-free") "GF" ""))) -}} {{- if $abbr -}} {{- $url := printf "%s?diet=%s" (strings.TrimSuffix "/" $base) $slug -}} {{ $abbr }} {{- end -}} {{- end -}} {{- end -}}
